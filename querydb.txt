CREATE TABLE accounts (
     id int NOT NULL AUTO_INCREMENT,
     PRIMARY KEY (id),
     firstname CHAR(30) NOT NULL,
     lastname varchar(50) NULL,
     username nvarchar(255),
     password char(68),
     phone char(11),
     role_id int,
     FOREIGN KEY (role_id) REFERENCES roles(id)
);

create table devideToken(
	id int NOT NULL AUTO_INCREMENT,
	PRIMARY KEY (id),
	accountId int,
	FOREIGN KEY (accountId) REFERENCES accounts(id),
	token nvarchar(255)
);

CREATE TABLE tournament(
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    tournamentName nvarchar(50),
    startDate date,
    endDate date,
    status BOOLEAN not null
 );

 CREATE TABLE round(
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    tournamentId int,
    FOREIGN KEY (tournamentId) REFERENCES tournament(id),
    roundName nvarchar(25),
    startDate date,
    endDate date,
    status BOOLEAN not null
 );


CREATE TABLE stadium(
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    stadiumName nvarchar(50),
    location nvarchar(255),
    capacity int,
    img nvarchar(255)
 );
 
 

 CREATE TABLE area (
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    stadiumId int,
    FOREIGN KEY (stadiumId) REFERENCES stadium(id),
    areaName char(10),
    capacity int
 );
 


 CREATE TABLE club (
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    stadiumId int null,
    FOREIGN KEY (stadiumId) REFERENCES stadium(id),
    clubName nvarchar(100),
    img nvarchar(255),
    country nvarchar(25)
 );

 
 CREATE TABLE matches (
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    roundId int,
    FOREIGN KEY (roundId) REFERENCES round(id),
    stadiumId int,
	FOREIGN KEY (stadiumId) REFERENCES stadium(id),
    clubHomeId int, 
    FOREIGN KEY (clubHomeId) REFERENCES club(id),
    clubVisitorId int,
    FOREIGN KEY (clubVisitorId) REFERENCES club(id),
    timeStart datetime,
    status BOOLEAN not null
 );
 

 CREATE TABLE systemTicketType (
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    systemTicketTypeName nvarchar(25)
 );


CREATE TABLE ticket (
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    matchId int,
    FOREIGN KEY (matchId) REFERENCES matches(id),
    areaId int,
    FOREIGN KEY (areaId) REFERENCES area(id),
    price double,
    amount int
 );
 
 CREATE TABLE paymentMethod (
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    paymentMethodName nvarchar(25)
 );
 
 CREATE TABLE orders (
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    accountId int,
    FOREIGN KEY (accountId) REFERENCES accounts(id),
    orderDate datetime,
    total double,
    status int
 );
 
  CREATE TABLE orderDetail (
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    orderId int,
    FOREIGN KEY (orderId) REFERENCES orders(id),
    ticketId int,
    FOREIGN KEY (ticketId) REFERENCES ticket(id),
    quantity int
 );
 

  CREATE TABLE invoice (
	id int NOT NULL AUTO_INCREMENT,
    primary key(id),
    orderDetailId int,
    FOREIGN KEY (orderDetailId) REFERENCES orderDetail(id),
    accountId int,
    FOREIGN KEY (accountId) REFERENCES accounts(id),
	phone nvarchar(25),
    total double,
    status nvarchar(25),
    currentDate datetime,
    paymentMethodId int,
	FOREIGN KEY (paymentMethodId) REFERENCES paymentMethod(id)
 );
